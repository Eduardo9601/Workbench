SELECT DISTINCT COD_CONTRATO, DES_PESSOA, DES_UNIDADE, DATA_INI_AFAST, DATA_FIM_AFAST, STATUS_AFAST
FROM V_DADOS_COLAB_AVT
WHERE STATUS = 0
AND DATA_DEMISSAO IS NULL
AND STATUS_AFAST NOT IN ('AFASTADO')
AND COD_UNIDADE IN ('488', '7589', '631', '413', '7412', '7093', '7022', '7041', '566', '7173', '316', '527', '7591', '7491', '384')
  OR COD_CONTRATO IN (385769, 394936, 393301, 387992, 391794, 391013, 393737, 394537, 391797, 394807)
